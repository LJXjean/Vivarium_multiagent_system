Insect:

sequenceDiagram
    actor Insect as Insect
    participant Env as Environment
    participant Food as Food Source
    participant Partner as Mating Partner
    participant Predator as Predator

    activate Insect
    Insect ->>+ Env: Request Food Location
    Env -->>- Insect: Return Food Location
    Insect ->>+ Food: Move to Food Source
    Food -->>- Insect: Eating
    Insect ->>+ Env: Search for Mating Partner
    Env -->>- Insect: Return Partner Location
    Insect ->>+ Partner: Mating
    Partner -->>- Insect: Mating Successful
    Insect ->> Insect: Random Movement
    loop Avoiding Predators
        Insect ->>+ Env: Request Predator Info
        Env -->>- Insect: Return Predator Info
        Insect ->> Insect: Evade if Necessary
    end
    deactivate Insect

Plante:

sequenceDiagram
    actor Plant as Plant
    participant Env as Environment

    activate Plant
    loop Daily Cycle
        Env ->>+ Plant: Update Environment (Temperature, Humidity, Light)
        Plant -->>- Env: Photosynthesis and Growth
        Plant ->> Plant: Random Seed Dispersal
    end
    Plant ->> Plant: Aging
    deactivate Plant
